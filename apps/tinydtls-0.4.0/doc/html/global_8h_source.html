100 static inline int
101 _dtls_address_equals_impl(const session_t *a,
102  const session_t *b) {
103  if (a->ifindex != b->ifindex ||
104  a->size != b->size || a->addr.sa.sa_family != b->addr.sa.sa_family)
105  return 0;
106 
107  /* need to compare only relevant parts of sockaddr_in6 */
108  switch (a->addr.sa.sa_family) {
109  case AF_INET:
110  return
111  a->addr.sin.sin_port == b->addr.sin.sin_port &&
112  memcmp(&a->addr.sin.sin_addr, &b->addr.sin.sin_addr,
113  sizeof(struct in_addr)) == 0;
114  case AF_INET6:
115  return a->addr.sin6.sin6_port == b->addr.sin6.sin6_port &&
116  memcmp(&a->addr.sin6.sin6_addr, &b->addr.sin6.sin6_addr,
117  sizeof(struct in6_addr)) == 0;
118  default: /* fall through and signal error */
119  ;
120  }
121  return 0;
122 }
123 #endif /* session_t */
